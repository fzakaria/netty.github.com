60  /**
61   * Adds <a href="http://en.wikipedia.org/wiki/Transport_Layer_Security">SSL
62   * &middot; TLS</a> and StartTLS support to a {@link Channel}.  Please refer
63   * to the <strong>"SecureChat"</strong> example in the distribution or the web
64   * site for the detailed usage.
65   *
66   * <h3>Beginning the handshake</h3>
67   * <p>
68   * You must make sure not to write a message while the handshake is in progress unless you are
69   * renegotiating.  You will be notified by the {@link Future} which is
70   * returned by the {@link #handshakeFuture()} method when the handshake
71   * process succeeds or fails.
72   * <p>
73   * Beside using the handshake {@link ChannelFuture} to get notified about the completation of the handshake it's
74   * also possible to detect it by implement the
75   * {@link ChannelHandler#userEventTriggered(ChannelHandlerContext, Object)}
76   * method and check for a {@link SslHandshakeCompletionEvent}.
77   *
78   * <h3>Handshake</h3>
79   * <p>
80   * The handshake will be automaticly issued for you once the {@link Channel} is active and
81   * {@link SSLEngine#getUseClientMode()} returns {@code true}.
82   * So no need to bother with it by your self.
83   *
84   * <h3>Closing the session</h3>
85   * <p>
86   * To close the SSL session, the {@link #close()} method should be
87   * called to send the {@code close_notify} message to the remote peer.  One
88   * exception is when you close the {@link Channel} - {@link SslHandler}
89   * intercepts the close request and send the {@code close_notify} message
90   * before the channel closure automatically.  Once the SSL session is closed,
91   * it is not reusable, and consequently you should create a new
92   * {@link SslHandler} with a new {@link SSLEngine} as explained in the
93   * following section.
94   *
95   * <h3>Restarting the session</h3>
96   * <p>
97   * To restart the SSL session, you must remove the existing closed
98   * {@link SslHandler} from the {@link ChannelPipeline}, insert a new
99   * {@link SslHandler} with a new {@link SSLEngine} into the pipeline,
100  * and start the handshake process as described in the first section.
101  *
102  * <h3>Implementing StartTLS</h3>
103  * <p>
104  * <a href="http://en.wikipedia.org/wiki/STARTTLS">StartTLS</a> is the
105  * communication pattern that secures the wire in the middle of the plaintext
106  * connection.  Please note that it is different from SSL &middot; TLS, that
107  * secures the wire from the beginning of the connection.  Typically, StartTLS
108  * is composed of three steps:
109  * <ol>
110  * <li>Client sends a StartTLS request to server.</li>
111  * <li>Server sends a StartTLS response to client.</li>
112  * <li>Client begins SSL handshake.</li>
113  * </ol>
114  * If you implement a server, you need to:
115  * <ol>
116  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
117  *     to {@code true},</li>
118  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
119  * <li>write a StartTLS response.</li>
120  * </ol>
121  * Please note that you must insert {@link SslHandler} <em>before</em> sending
122  * the StartTLS response.  Otherwise the client can send begin SSL handshake
123  * before {@link SslHandler} is inserted to the {@link ChannelPipeline}, causing
124  * data corruption.
125  * <p>
126  * The client-side implementation is much simpler.
127  * <ol>
128  * <li>Write a StartTLS request,</li>
129  * <li>wait for the StartTLS response,</li>
130  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
131  *     to {@code false},</li>
132  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
133  * <li>Initiate SSL handshake.</li>
134  * </ol>
135  *
136  * <h3>Known issues</h3>
137  * <p>
138  * Because of a known issue with the current implementation of the SslEngine that comes
139  * with Java it may be possible that you see blocked IO-Threads while a full GC is done.
140  * <p>
141  * So if you are affected you can workaround this problem by adjust the cache settings
142  * like shown below:
143  *
144  * <pre>
145  *     SslContext context = ...;
146  *     context.getServerSessionContext().setSessionCacheSize(someSaneSize);
147  *     context.getServerSessionContext().setSessionTime(someSameTimeout);
148  * </pre>
149  * <p>
150  * What values to use here depends on the nature of your application and should be set
151  * based on monitoring and debugging of it.
152  * For more details see
153  * <a href="https://github.com/netty/netty/issues/832">#832</a> in our issue tracker.
154  */

289         final ChannelHandlerContext ctx = this.ctx;

294                 try {
295                     write(ctx, Unpooled.EMPTY_BUFFER, future);

297                 } catch (Exception e) {
298                     if (!future.tryFailure(e)) {

300                     }
301                 }
302             }
303         });

368     @Override
369     public void flush(ChannelHandlerContext ctx) throws Exception {
370         // Do not encrypt the first write request if this handler is
371         // created with startTLS flag turned on.
372         if (startTls && !sentFirstMessage) {
373             sentFirstMessage = true;
374             for (;;) {
375                 PendingWrite pendingWrite = pendingUnencryptedWrites.poll();
376                 if (pendingWrite == null) {
377                     break;
378                 }
379                 ctx.write(pendingWrite.msg(), (ChannelPromise) pendingWrite.recycleAndGet());
380             }
381             ctx.flush();
382             return;
383         }
384         if (pendingUnencryptedWrites.isEmpty()) {
385             pendingUnencryptedWrites.add(PendingWrite.newInstance(Unpooled.EMPTY_BUFFER, null));
386         }
387         wrap(ctx, false);
388         ctx.flush();
389     }

391     private void wrap(ChannelHandlerContext ctx, boolean inUnwrap) throws SSLException {
392         ByteBuf out = null;
393         ChannelPromise promise = null;
394         try {
395             for (;;) {
396                 PendingWrite pending = pendingUnencryptedWrites.peek();
397                 if (pending == null) {
398                     break;
399                 }
400                 if (out == null) {
401                     out = ctx.alloc().buffer(maxPacketBufferSize);
402                 }
403 
404                 if (!(pending.msg() instanceof ByteBuf)) {
405                     ctx.write(pending.msg(), (ChannelPromise) pending.recycleAndGet());
406                     pendingUnencryptedWrites.remove();
407                     continue;
408                 }
409 
410                 ByteBuf buf = (ByteBuf) pending.msg();
411                 SSLEngineResult result = wrap(engine, buf, out);
412 
413                 if (!buf.isReadable()) {
414                     buf.release();
415                     promise = (ChannelPromise) pending.recycleAndGet();
416                     pendingUnencryptedWrites.remove();
417                 } else {
418                     promise = null;
419                 }
420 
421                 if (result.getStatus() == Status.CLOSED) {
422                     // SSLEngine has been closed already.
423                     // Any further write attempts should be denied.
424                     for (;;) {
425                         PendingWrite w = pendingUnencryptedWrites.poll();
426                         if (w == null) {
427                             break;
428                         }
429                         w.failAndRecycle(SSLENGINE_CLOSED);
430                     }
431                     return;
432                 } else {
433                     switch (result.getHandshakeStatus()) {
434                         case NEED_TASK:
435                             runDelegatedTasks();
436                             break;
437                         case FINISHED:
438                             setHandshakeSuccess();
439                             // deliberate fall-through
440                         case NOT_HANDSHAKING:
441                         case NEED_WRAP:
442                             finishWrap(ctx, out, promise, inUnwrap);
443                             promise = null;
444                             out = null;
445                             break;
446                         case NEED_UNWRAP:
447                             return;
448                         default:
449                             throw new IllegalStateException("Unknown handshake status: " + result.getHandshakeStatus());
450                     }
451                 }
452             }
453         } catch (SSLException e) {
454             setHandshakeFailure(e);
455             throw e;
456         } finally {
457             finishWrap(ctx, out, promise, inUnwrap);
458         }
459     }

461     private void finishWrap(ChannelHandlerContext ctx, ByteBuf out, ChannelPromise promise, boolean inUnwrap) {
462         if (out == null) {
463             out = Unpooled.EMPTY_BUFFER;
464         } else if (!out.isReadable()) {
465             out.release();
466             out = Unpooled.EMPTY_BUFFER;
467         }
468 
469         if (promise != null) {
470             ctx.write(out, promise);
471         } else {
472             ctx.write(out);
473         }
474 
475         if (inUnwrap) {
476             needsFlush = true;
477         }
478     }

480     private void wrapNonAppData(ChannelHandlerContext ctx, boolean inUnwrap) throws SSLException {
481         ByteBuf out = null;
482         try {
483             for (;;) {
484                 if (out == null) {
485                     out = ctx.alloc().buffer(maxPacketBufferSize);
486                 }
487                 SSLEngineResult result = wrap(engine, Unpooled.EMPTY_BUFFER, out);
488 
489                 if (result.bytesProduced() > 0) {
490                     ctx.write(out);
491                     if (inUnwrap) {
492                         needsFlush = true;
493                     }
494                     out = null;
495                 }
496 
497                 switch (result.getHandshakeStatus()) {
498                     case FINISHED:
499                         setHandshakeSuccess();
500                         break;
501                     case NEED_TASK:
502                         runDelegatedTasks();
503                         break;
504                     case NEED_UNWRAP:
505                         if (!inUnwrap) {
506                             unwrap(ctx);
507                         }
508                         break;
509                     case NEED_WRAP:
510                         break;
511                     case NOT_HANDSHAKING:
512                         // Workaround for TLS False Start problem reported at:
513                         // https://github.com/netty/netty/issues/1108#issuecomment-14266970
514                         if (!inUnwrap) {
515                             unwrap(ctx);
516                         }
517                         break;
518                     default:
519                         throw new IllegalStateException("Unknown handshake status: " + result.getHandshakeStatus());
520                 }
521 
522                 if (result.bytesProduced() == 0) {
523                     break;
524                 }
525             }
526         } catch (SSLException e) {
527             setHandshakeFailure(e);
528             throw e;
529         }  finally {
530             if (out != null) {
531                 out.release();
532             }
533         }
534     }

536     private SSLEngineResult wrap(SSLEngine engine, ByteBuf in, ByteBuf out) throws SSLException {
537         ByteBuffer in0 = in.nioBuffer();
538         for (;;) {
539             ByteBuffer out0 = out.nioBuffer(out.writerIndex(), out.writableBytes());
540             SSLEngineResult result = engine.wrap(in0, out0);
541             in.skipBytes(result.bytesConsumed());
542             out.writerIndex(out.writerIndex() + result.bytesProduced());
543 
544             switch (result.getStatus()) {
545                 case BUFFER_OVERFLOW:
546                     out.ensureWritable(maxPacketBufferSize);
547                     break;
548                 default:
549                     return result;
550             }
551         }
552     }

562     @Override
563     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
564         if (ignoreException(cause)) {
565             // It is safe to ignore the 'connection reset by peer' or
566             // 'broken pipe' error after sending close_notify.
567             if (logger.isDebugEnabled()) {
568                 logger.debug(
569                         "Swallowing a harmless 'connection reset by peer / broken pipe' error that occurred " +
570                                 "while writing close_notify in response to the peer's close_notify", cause);
571             }
572 
573             // Close the connection explicitly just in case the transport
574             // did not close the connection automatically.
575             if (ctx.channel().isActive()) {
576                 ctx.close();
577             }
578         } else {
579             ctx.fireExceptionCaught(cause);
580         }
581     }

583     /**
584      * Checks if the given {@link Throwable} can be ignore and just "swallowed"
585      *
586      * When an ssl connection is closed a close_notify message is sent.
587      * After that the peer also sends close_notify however, it's not mandatory to receive
588      * the close_notify. The party who sent the initial close_notify can close the connection immediately
589      * then the peer will get connection reset error.
590      *
591      */
592     private boolean ignoreException(Throwable t) {
593         if (!(t instanceof SSLException) && t instanceof IOException && sslCloseFuture.isDone()) {
594             String message = String.valueOf(t.getMessage()).toLowerCase();
595 
596             // first try to match connection reset / broke peer based on the regex. This is the fastest way
597             // but may fail on different jdk impls or OS's
598             if (IGNORABLE_ERROR_MESSAGE.matcher(message).matches()) {
599                 return true;
600             }
601 
602             // Inspect the StackTraceElements to see if it was a connection reset / broken pipe or not
603             StackTraceElement[] elements = t.getStackTrace();
604             for (StackTraceElement element: elements) {
605                 String classname = element.getClassName();
606                 String methodname = element.getMethodName();
607 
608                 // skip all classes that belong to the io.netty package
609                 if (classname.startsWith("io.netty.")) {
610                     continue;
611                 }
612 
613                 // check if the method name is read if not skip it
614                 if (!"read".equals(methodname)) {
615                     continue;
616                 }
617 
618                 // This will also match against SocketInputStream which is used by openjdk 7 and maybe
619                 // also others
620                 if (IGNORABLE_CLASS_IN_STACK.matcher(classname).matches()) {
621                     return true;
622                 }
623 
624                 try {
625                     // No match by now.. Try to load the class via classloader and inspect it.
626                     // This is mainly done as other JDK implementations may differ in name of
627                     // the impl.
628                     Class<?> clazz = getClass().getClassLoader().loadClass(classname);
629 
630                     if (SocketChannel.class.isAssignableFrom(clazz)
631                             || DatagramChannel.class.isAssignableFrom(clazz)) {
632                         return true;
633                     }
634 
635                     // also match against SctpChannel via String matching as it may not present.
636                     if (PlatformDependent.javaVersion() >= 7
637                             && "com.sun.nio.sctp.SctpChannel".equals(clazz.getSuperclass().getName())) {
638                         return true;
639                     }
640                 } catch (ClassNotFoundException e) {
641                     // This should not happen just ignore
642                 }
643             }
644         }
645 
646         return false;
647     }

649     /**
650      * Returns {@code true} if the given {@link ByteBuf} is encrypted. Be aware that this method
651      * will not increase the readerIndex of the given {@link ByteBuf}.
652      *
653      * @param   buffer
654      *                  The {@link ByteBuf} to read from. Be aware that it must have at least 5 bytes to read,
655      *                  otherwise it will throw an {@link IllegalArgumentException}.
656      * @return encrypted
657      *                  {@code true} if the {@link ByteBuf} is encrypted, {@code false} otherwise.
658      * @throws IllegalArgumentException
659      *                  Is thrown if the given {@link ByteBuf} has not at least 5 bytes to read.
660      */
661     public static boolean isEncrypted(ByteBuf buffer) {
662         if (buffer.readableBytes() < 5) {
663             throw new IllegalArgumentException("buffer must have at least 5 readable bytes");
664         }
665         return getEncryptedPacketLength(buffer, buffer.readerIndex()) != -1;
666     }

668     /**
669      * Return how much bytes can be read out of the encrypted data. Be aware that this method will not increase
670      * the readerIndex of the given {@link ByteBuf}.
671      *
672      * @param   buffer
673      *                  The {@link ByteBuf} to read from. Be aware that it must have at least 5 bytes to read,
674      *                  otherwise it will throw an {@link IllegalArgumentException}.
675      * @return length
676      *                  The length of the encrypted packet that is included in the buffer. This will
677      *                  return {@code -1} if the given {@link ByteBuf} is not encrypted at all.
678      * @throws IllegalArgumentException
679      *                  Is thrown if the given {@link ByteBuf} has not at least 5 bytes to read.
680      */
681     private static int getEncryptedPacketLength(ByteBuf buffer, int offset) {
682         int packetLength = 0;
683 
684         // SSLv3 or TLS - Check ContentType
685         boolean tls;
686         switch (buffer.getUnsignedByte(offset)) {
687             case 20:  // change_cipher_spec
688             case 21:  // alert
689             case 22:  // handshake
690             case 23:  // application_data
691                 tls = true;
692                 break;
693             default:
694                 // SSLv2 or bad data
695                 tls = false;
696         }
697 
698         if (tls) {
699             // SSLv3 or TLS - Check ProtocolVersion
700             int majorVersion = buffer.getUnsignedByte(offset + 1);
701             if (majorVersion == 3) {
702                 // SSLv3 or TLS
703                 packetLength = buffer.getUnsignedShort(offset + 3) + 5;
704                 if (packetLength <= 5) {
705                     // Neither SSLv3 or TLSv1 (i.e. SSLv2 or bad data)
706                     tls = false;
707                 }
708             } else {
709                 // Neither SSLv3 or TLSv1 (i.e. SSLv2 or bad data)
710                 tls = false;
711             }
712         }
713 
714         if (!tls) {
715             // SSLv2 or bad data - Check the version
716             boolean sslv2 = true;
717             int headerLength = (buffer.getUnsignedByte(offset) & 0x80) != 0 ? 2 : 3;
718             int majorVersion = buffer.getUnsignedByte(offset + headerLength + 1);
719             if (majorVersion == 2 || majorVersion == 3) {
720                 // SSLv2
721                 if (headerLength == 2) {
722                     packetLength = (buffer.getShort(offset) & 0x7FFF) + 2;
723                 } else {
724                     packetLength = (buffer.getShort(offset) & 0x3FFF) + 3;
725                 }
726                 if (packetLength <= headerLength) {
727                     sslv2 = false;
728                 }
729             } else {
730                 sslv2 = false;
731             }
732 
733             if (!sslv2) {
734                 return -1;
735             }
736         }
737         return packetLength;
738     }

740     @Override
741     protected void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws SSLException {
742         final int startOffset = in.readerIndex();
743         final int endOffset = in.writerIndex();
744         int offset = startOffset;
745 
746         // If we calculated the length of the current SSL record before, use that information.
747         if (packetLength > 0) {
748             if (endOffset - startOffset < packetLength) {
749                 return;
750             } else {
751                 offset += packetLength;
752                 packetLength = 0;
753             }
754         }
755 
756         boolean nonSslRecord = false;
757 
758         for (;;) {
759             final int readableBytes = endOffset - offset;
760             if (readableBytes < 5) {
761                 break;
762             }
763 
764             final int packetLength = getEncryptedPacketLength(in, offset);
765             if (packetLength == -1) {
766                 nonSslRecord = true;
767                 break;
768             }
769 
770             assert packetLength > 0;
771 
772             if (packetLength > readableBytes) {
773                 // wait until the whole packet can be read
774                 this.packetLength = packetLength;
775                 break;
776             }
777 
778             offset += packetLength;
779         }
780 
781         final int length = offset - startOffset;
782         if (length > 0) {
783             // The buffer contains one or more full SSL records.
784             // Slice out the whole packet so unwrap will only be called with complete packets.
785             // Also directly reset the packetLength. This is needed as unwrap(..) may trigger
786             // decode(...) again via:
787             // 1) unwrap(..) is called
788             // 2) wrap(...) is called from within unwrap(...)
789             // 3) wrap(...) calls unwrapLater(...)
790             // 4) unwrapLater(...) calls decode(...)
791             //
792             // See https://github.com/netty/netty/issues/1534
793             in.skipBytes(length);
794             ByteBuffer buffer = in.nioBuffer(startOffset, length);
795             unwrap(ctx, buffer, out);
796         }
797 
798         if (nonSslRecord) {
799             // Not an SSL/TLS packet
800             NotSslRecordException e = new NotSslRecordException(
801                     "not an SSL/TLS record: " + ByteBufUtil.hexDump(in));
802             in.skipBytes(in.readableBytes());
803             ctx.fireExceptionCaught(e);
804             setHandshakeFailure(e);
805         }
806     }

817     private void unwrap(ChannelHandlerContext ctx) throws SSLException {
818         RecyclableArrayList out = RecyclableArrayList.newInstance();
819         try {
820             unwrap(ctx, Unpooled.EMPTY_BUFFER.nioBuffer(), out);
821             final int size = out.size();
822             for (int i = 0; i < size; i++) {
823                 ctx.fireChannelRead(out.get(i));
824             }
825         } finally {
826             out.recycle();
827         }

830     private void unwrap(ChannelHandlerContext ctx, ByteBuffer packet, List<Object> out) throws SSLException {
831         boolean wrapLater = false;
832         int totalProduced = 0;
833         try {
834             for (;;) {
835                 if (decodeOut == null) {
836                     decodeOut = ctx.alloc().buffer(packet.remaining());
837                 }
838 
839                 final SSLEngineResult result = unwrap(engine, packet, decodeOut);
840                 final Status status = result.getStatus();
841                 final HandshakeStatus handshakeStatus = result.getHandshakeStatus();
842                 final int produced = result.bytesProduced();
843                 final int consumed = result.bytesConsumed();
844 
845                 totalProduced += produced;
846                 if (status == Status.CLOSED) {
847                     // notify about the CLOSED state of the SSLEngine. See #137
848                     sslCloseFuture.trySuccess(ctx.channel());
849                     break;
850                 }
851 
852                 switch (handshakeStatus) {
853                     case NEED_UNWRAP:
854                         break;
855                     case NEED_WRAP:
856                         wrapNonAppData(ctx, true);
857                         break;
858                     case NEED_TASK:
859                         runDelegatedTasks();
860                         break;
861                     case FINISHED:
862                         setHandshakeSuccess();
863                         wrapLater = true;
864                         continue;
865                     case NOT_HANDSHAKING:
866                         break;
867                     default:
868                         throw new IllegalStateException("Unknown handshake status: " + handshakeStatus);
869                 }
870 
871                 if (status == Status.BUFFER_UNDERFLOW || consumed == 0 && produced == 0) {
872                     break;
873                 }
874             }
875 
876             if (wrapLater) {
877                 wrap(ctx, true);
878             }
879         } catch (SSLException e) {
880             setHandshakeFailure(e);
881             throw e;
882         } finally {
883             if (totalProduced > 0) {
884                 ByteBuf decodeOut = this.decodeOut;
885                 this.decodeOut = null;
886                 out.add(decodeOut);
887             }
888         }
889     }

891     private static SSLEngineResult unwrap(SSLEngine engine, ByteBuffer in, ByteBuf out) throws SSLException {
892         int overflows = 0;
893         for (;;) {
894             ByteBuffer out0 = out.nioBuffer(out.writerIndex(), out.writableBytes());
895             SSLEngineResult result = engine.unwrap(in, out0);
896             out.writerIndex(out.writerIndex() + result.bytesProduced());
897             switch (result.getStatus()) {
898                 case BUFFER_OVERFLOW:
899                     int max = engine.getSession().getApplicationBufferSize();
900                     switch (overflows ++) {
901                         case 0:
902                             out.ensureWritable(Math.min(max, in.remaining()));
903                             break;
904                         default:
905                             out.ensureWritable(max);
906                     }
907                     break;
908                 default:
909                     return result;
910             }
911         }
912     }

943         engine.closeOutbound();

1007     private Future<Channel> handshake() {
1008         final ScheduledFuture<?> timeoutFuture;
1009         if (handshakeTimeoutMillis > 0) {
1010             timeoutFuture = ctx.executor().schedule(new Runnable() {
1011                 @Override
1012                 public void run() {
1013                     if (handshakePromise.isDone()) {
1014                         return;
1015                     }
1016                     notifyHandshakeFailure(HANDSHAKE_TIMED_OUT);
1017                 }
1018             }, handshakeTimeoutMillis, TimeUnit.MILLISECONDS);
1019         } else {
1020             timeoutFuture = null;
1021         }
1022 
1023         handshakePromise.addListener(new GenericFutureListener<Future<Channel>>() {
1024             @Override
1025             public void operationComplete(Future<Channel> f) throws Exception {
1026                 if (timeoutFuture != null) {
1027                     timeoutFuture.cancel(false);
1028                 }
1029             }
1030         });
1031         try {
1032             engine.beginHandshake();
1033             wrapNonAppData(ctx, false);
1034             ctx.flush();
1035         } catch (Exception e) {
1036             notifyHandshakeFailure(e);
1037         }
1038         return handshakePromise;
1039     }

1061     private void safeClose(
1062             final ChannelHandlerContext ctx, ChannelFuture flushFuture,
1063             final ChannelPromise promise) {
1064         if (!ctx.channel().isActive()) {
1065             ctx.close(promise);
1066             return;

1068 
1069         final ScheduledFuture<?> timeoutFuture;

1081         } else {
1082             timeoutFuture = null;

1086         flushFuture.addListener(new ChannelFutureListener() {
1087             @Override
1088             public void operationComplete(ChannelFuture f)
1089                     throws Exception {
1090                 if (timeoutFuture != null) {
1091                     timeoutFuture.cancel(false);
1092                 }
1093                 if (ctx.channel().isActive()) {
