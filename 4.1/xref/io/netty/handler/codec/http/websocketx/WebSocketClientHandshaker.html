141     /**
142      * Begins the opening handshake
143      *
144      * @param channel
145      *            Channel
146      * @param promise
147      *            the {@link ChannelPromise} to be notified when the opening handshake is sent
148      */
149     public final ChannelFuture handshake(Channel channel, final ChannelPromise promise) {
150         FullHttpRequest request =  newHandshakeRequest();
151 
152         HttpResponseDecoder decoder = channel.pipeline().get(HttpResponseDecoder.class);
153         if (decoder == null) {
154             HttpClientCodec codec = channel.pipeline().get(HttpClientCodec.class);
155             if (codec == null) {
156                promise.setFailure(new IllegalStateException("ChannelPipeline does not contain " +
157                        "a HttpResponseDecoder or HttpClientCodec"));
158                return promise;
159             }
160         }
161 
162         channel.writeAndFlush(request).addListener(new ChannelFutureListener() {
163             @Override
164             public void operationComplete(ChannelFuture future) {
165                 if (future.isSuccess()) {
166                     ChannelPipeline p = future.channel().pipeline();
167                     ChannelHandlerContext ctx = p.context(HttpRequestEncoder.class);
168                     if (ctx == null) {
169                         ctx = p.context(HttpClientCodec.class);
170                     }
171                     if (ctx == null) {
172                         promise.setFailure(new IllegalStateException("ChannelPipeline does not contain " +
173                                 "a HttpRequestEncoder or HttpClientCodec"));
174                         return;
175                     }
176                     p.addAfter(ctx.name(), "ws-encoder", newWebSocketEncoder());
177 
178                     promise.setSuccess();
179                 } else {
180                     promise.setFailure(future.cause());
181                 }
182             }
183         });
184         return promise;
185     }

187     /**

192     /**
193      * Validates and finishes the opening handshake initiated by {@link #handshake}}.
194      *
195      * @param channel
196      *            Channel
197      * @param response
198      *            HTTP response containing the closing handshake details
199      */
200     public final void finishHandshake(Channel channel, FullHttpResponse response) {
201         verify(response);
202         setActualSubprotocol(response.headers().get(HttpHeaders.Names.SEC_WEBSOCKET_PROTOCOL));
203         setHandshakeComplete();
204 
205         ChannelPipeline p = channel.pipeline();
206         // Remove decompressor from pipeline if its in use
207         HttpContentDecompressor decompressor = p.get(HttpContentDecompressor.class);
208         if (decompressor != null) {
209             p.remove(decompressor);
210         }
211 
212         ChannelHandlerContext ctx = p.context(HttpResponseDecoder.class);
213         if (ctx == null) {
214             ctx = p.context(HttpClientCodec.class);
215             if (ctx == null) {
216                 throw new IllegalStateException("ChannelPipeline does not contain " +
217                         "a HttpRequestEncoder or HttpClientCodec");
218             }
219             p.replace(ctx.name(), "ws-decoder", newWebsocketDecoder());
220         } else {
221             if (p.get(HttpRequestEncoder.class) != null) {
222                 p.remove(HttpRequestEncoder.class);
223             }
224             p.replace(ctx.name(),
225                     "ws-decoder", newWebsocketDecoder());
226         }
227     }
