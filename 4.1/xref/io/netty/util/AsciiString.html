32  /**
33   * A string which has been encoded into a character encoding whose character always takes a single byte, similarly to
34   * ASCII. It internally keeps its content in a byte array unlike {@link String}, which uses a character array, for
35   * reduced memory footprint and faster data transfer from/to byte-based data structures such as a byte array and
36   * {@link ByteBuffer}. It is often used in conjunction with {@link TextHeaders}.
37   */
38  public final class AsciiString extends ByteString implements CharSequence, Comparable<CharSequence> {
39  
40      private static final char MAX_CHAR_VALUE = 255;
41      public static final AsciiString EMPTY_STRING = new AsciiString("");

44              new HashingStrategy<CharSequence>() {
45          @Override
46          public int hashCode(CharSequence o) {

48          }
49  
50          @Override
51          public boolean equals(CharSequence a, CharSequence b) {
52              return AsciiString.contentEqualsIgnoreCase(a, b);

64              return AsciiString.contentEquals(a, b);

169     /**
170      * Compares the specified string to this string using the ASCII values of the characters. Returns 0 if the strings
171      * contain the same characters in the same order. Returns a negative integer if the first non-equal character in
172      * this string has an ASCII value which is less than the ASCII value of the character at the same position in the
173      * specified string, or if this string is a prefix of the specified string. Returns a positive integer if the first
174      * non-equal character in this string has a ASCII value which is greater than the ASCII value of the character at
175      * the same position in the specified string, or if the specified string is a prefix of this string.
176      *
177      * @param string the string to compare.
178      * @return 0 if the strings are equal, a negative integer if this string is before the specified string, or a
179      *         positive integer if this string is after the specified string.
180      * @throws NullPointerException if {@code string} is {@code null}.
181      */
182     @Override
183     public int compareTo(CharSequence string) {
184         if (this == string) {
185             return 0;
186         }
187 
188         int result;
189         int length1 = length();
190         int length2 = string.length();
191         int minLength = Math.min(length1, length2);
192         for (int i = 0, j = arrayOffset(); i < minLength; i++, j++) {
193             result = (char) (value[j] & 0xFF) - string.charAt(i);
194             if (result != 0) {
195                 return result;
196             }
197         }
198 
199         return length1 - length2;
200     }

202     /**
203      * Concatenates this string and the specified string.
204      *
205      * @param string the string to concatenate
206      * @return a new string which is the concatenation of this string and the specified string.
207      */
208     public AsciiString concat(CharSequence string) {
209         int thisLen = length();
210         int thatLen = string.length();
211         if (thatLen == 0) {
212             return this;
213         }
214 
215         if (string instanceof AsciiString) {
216             AsciiString that = (AsciiString) string;
217             if (isEmpty()) {
218                 return that;
219             }
220 
221             byte[] newValue = new byte[thisLen + thatLen];
222             System.arraycopy(value, arrayOffset(), newValue, 0, thisLen);
223             System.arraycopy(that.value, that.arrayOffset(), newValue, thisLen, thatLen);
224             return new AsciiString(newValue, false);
225         }
226 
227         if (isEmpty()) {
228             return new AsciiString(string);
229         }
230 
231         byte[] newValue = new byte[thisLen + thatLen];
232         System.arraycopy(value, arrayOffset(), newValue, 0, thisLen);
233         for (int i = thisLen, j = 0; i < newValue.length; i++, j++) {
234             newValue[i] = c2b(string.charAt(j));
235         }
236 
237         return new AsciiString(newValue, false);
238     }

253      * Compares the specified string to this string ignoring the case of the characters and returns true if they are
254      * equal.
255      *
256      * @param string the string to compare.
257      * @return {@code true} if the specified string is equal to this string, {@code false} otherwise.

259     public boolean contentEqualsIgnoreCase(CharSequence string) {
260         if (string == null || string.length() != length()) {
261             return false;
262         }
263 
264         if (string.getClass() == AsciiString.class) {
265             AsciiString rhs = (AsciiString) string;
266             for (int i = arrayOffset(), j = rhs.arrayOffset(); i < length(); ++i, ++j) {
267                 if (!equalsIgnoreCase(value[i], value[j])) {
268                     return false;
269                 }
270             }
271             return true;
272         }
273 
274         for (int i = arrayOffset(), j = 0; i < length(); ++i, ++j) {
275             if (!equalsIgnoreCase((char) (value[i] & 0xFF), string.charAt(j))) {
276                 return false;
277             }
278         }
279         return true;
280     }

366     /**
367      * Searches in this string for the index of the specified string. The search for the string starts at the specified
368      * offset and moves towards the end of this string.
369      *
370      * @param subString the string to find.
371      * @param start the starting offset.
372      * @return the index of the first character of the specified string in this string, -1 if the specified string is
373      *         not a substring.
374      * @throws NullPointerException if {@code subString} is {@code null}.
375      */
376     public int indexOf(CharSequence subString, int start) {
377         if (start < 0) {
378             start = 0;
379         }
380 
381         final int thisLen = length();
382 
383         int subCount = subString.length();
384         if (subCount <= 0) {
385             return start < thisLen ? start : thisLen;
386         }
387         if (subCount > thisLen - start) {
388             return -1;
389         }
390 
391         final char firstChar = subString.charAt(0);
392         if (firstChar > MAX_CHAR_VALUE) {
393             return -1;
394         }
395         ByteProcessor IndexOfVisitor = new IndexOfProcessor((byte) firstChar);
396         try {
397             for (;;) {
398                 int i = forEachByte(start, thisLen - start, IndexOfVisitor);
399                 if (i == -1 || subCount + i > thisLen) {
400                     return -1; // handles subCount > count || start >= count
401                 }
402                 int o1 = i, o2 = 0;
403                 while (++o2 < subCount && (char) (value[++o1 + arrayOffset()] & 0xFF) == subString.charAt(o2)) {
404                     // Intentionally empty
405                 }
406                 if (o2 == subCount) {
407                     return i;
408                 }
409                 start = i + 1;
410             }
411         } catch (Exception e) {
412             PlatformDependent.throwException(e);
413             return -1;
414         }
415     }

431     /**
432      * Searches in this string for the index of the specified string. The search for the string starts at the specified
433      * offset and moves towards the beginning of this string.
434      *
435      * @param subString the string to find.
436      * @param start the starting offset.
437      * @return the index of the first character of the specified string in this string , -1 if the specified string is
438      *         not a substring.
439      * @throws NullPointerException if {@code subString} is {@code null}.
440      */
441     public int lastIndexOf(CharSequence subString, int start) {
442         final int thisLen = length();
443         final int subCount = subString.length();
444 
445         if (subCount > thisLen || start < 0) {
446             return -1;
447         }
448 
449         if (subCount <= 0) {
450             return start < thisLen ? start : thisLen;
451         }
452 
453         start = Math.min(start, thisLen - subCount);
454 
455         // count and subCount are both >= 1
456         final char firstChar = subString.charAt(0);
457         if (firstChar > MAX_CHAR_VALUE) {
458             return -1;
459         }
460         ByteProcessor IndexOfVisitor = new IndexOfProcessor((byte) firstChar);
461         try {
462             for (;;) {
463                 int i = forEachByteDesc(start, thisLen - start, IndexOfVisitor);
464                 if (i == -1) {
465                     return -1;
466                 }
467                 int o1 = i, o2 = 0;
468                 while (++o2 < subCount && (char) (value[++o1 + arrayOffset()] & 0xFF) == subString.charAt(o2)) {
469                     // Intentionally empty
470                 }
471                 if (o2 == subCount) {
472                     return i;
473                 }
474                 start = i - 1;
475             }
476         } catch (Exception e) {
477             PlatformDependent.throwException(e);
478             return -1;
479         }
480     }

482     /**
483      * Compares the specified string to this string and compares the specified range of characters to determine if they
484      * are the same.
485      *
486      * @param thisStart the starting offset in this string.
487      * @param string the string to compare.
488      * @param start the starting offset in the specified string.
489      * @param length the number of characters to compare.
490      * @return {@code true} if the ranges of characters are equal, {@code false} otherwise
491      * @throws NullPointerException if {@code string} is {@code null}.
492      */
493     public boolean regionMatches(int thisStart, CharSequence string, int start, int length) {
494         if (string == null) {
495             throw new NullPointerException("string");
496         }
497 
498         if (start < 0 || string.length() - start < length) {
499             return false;
500         }
501 
502         final int thisLen = length();
503         if (thisStart < 0 || thisLen - thisStart < length) {
504             return false;
505         }
506 
507         if (length <= 0) {
508             return true;
509         }
510 
511         final int thatEnd = start + length;
512         for (int i = start, j = thisStart + arrayOffset(); i < thatEnd; i++, j++) {
513             if ((char) (value[j] & 0xFF) != string.charAt(i)) {
514                 return false;
515             }
516         }
517         return true;
518     }

520     /**
521      * Compares the specified string to this string and compares the specified range of characters to determine if they
522      * are the same. When ignoreCase is true, the case of the characters is ignored during the comparison.
523      *
524      * @param ignoreCase specifies if case should be ignored.
525      * @param thisStart the starting offset in this string.
526      * @param string the string to compare.
527      * @param start the starting offset in the specified string.
528      * @param length the number of characters to compare.
529      * @return {@code true} if the ranges of characters are equal, {@code false} otherwise.
530      * @throws NullPointerException if {@code string} is {@code null}.
531      */
532     public boolean regionMatches(boolean ignoreCase, int thisStart, CharSequence string, int start, int length) {
533         if (!ignoreCase) {
534             return regionMatches(thisStart, string, start, length);
535         }
536 
537         if (string == null) {
538             throw new NullPointerException("string");
539         }
540 
541         final int thisLen = length();
542         if (thisStart < 0 || length > thisLen - thisStart) {
543             return false;
544         }
545         if (start < 0 || length > string.length() - start) {
546             return false;
547         }
548 
549         thisStart += arrayOffset();
550         final int thisEnd = thisStart + length;
551         while (thisStart < thisEnd) {
552             if (!equalsIgnoreCase((char) (value[thisStart++] & 0xFF), string.charAt(start++))) {
553                 return false;
554             }
555         }
556         return true;
557     }

559     /**
560      * Copies this string replacing occurrences of the specified character with another character.
561      *
562      * @param oldChar the character to replace.
563      * @param newChar the replacement character.
564      * @return a new string with occurrences of oldChar replaced by newChar.
565      */
566     public AsciiString replace(char oldChar, char newChar) {
567         if (oldChar > MAX_CHAR_VALUE) {
568             return this;
569         }
570 
571         final int index;
572         final byte oldCharByte = c2b(oldChar);
573         try {
574             index = forEachByte(new IndexOfProcessor(oldCharByte));
575         } catch (Exception e) {
576             PlatformDependent.throwException(e);
577             return this;
578         }
579         if (index == -1) {
580             return this;
581         }
582 
583         final byte newCharByte = c2b(newChar);
584         byte[] buffer = new byte[length()];
585         for (int i = 0, j = arrayOffset(); i < buffer.length; i++, j++) {
586             byte b = value[j];
587             if (b == oldCharByte) {
588                 b = newCharByte;
589             }
590             buffer[i] = b;
591         }
592 
593         return new AsciiString(buffer, false);
594     }

683      *
684      * @return a new string with characters {@code <= \\u0020} removed from the beginning and the end.
685      */
686     public AsciiString trim() {
687         int start = arrayOffset(), last = arrayOffset() + length();
688         int end = last;
689         while (start <= end && value[start] <= ' ') {
690             start++;
691         }
692         while (end >= start && value[end] <= ' ') {
693             end--;
694         }
695         if (start == 0 && end == last) {
696             return this;
697         }
698         return new AsciiString(value, start, end - start + 1, false);

701     /**
702      * Compares a {@code CharSequence} to this {@code String} to determine if their contents are equal.
703      *
704      * @param a the character sequence to compare to.
705      * @return {@code true} if equal, otherwise {@code false}
706      */
707     public boolean contentEquals(CharSequence a) {
708         if (this == a) {

1002     private static boolean contains(CharSequence a, CharSequence b, CharEqualityComparator cmp) {
1003         if (a == null || b == null || a.length() < b.length()) {
1004             return false;
1005         }
1006         if (b.length() == 0) {
1007             return true;
1008         }
1009         int bStart = 0;
1010         for (int i = 0; i < a.length(); ++i) {
1011             if (cmp.equals(b.charAt(bStart), a.charAt(i))) {
1012                 // If b is consumed then true.
1013                 if (++bStart == b.length()) {
1014                     return true;
1015                 }
1016             } else if (a.length() - i < b.length()) {
1017                 // If there are not enough characters left in a for b to be contained, then false.
1018                 return false;
1019             } else {
1020                 bStart = 0;
1021             }
1022         }
1023         return false;
1024     }
