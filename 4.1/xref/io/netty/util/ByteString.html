29  /**
30   * The primary use case for this class is to function as an immutable array of bytes. For performance reasons this
31   * class supports sharing memory with external arrays, and also direct access to the underlying memory via
32   * {@link #array()}. Care must be taken when directly accessing the memory as that may invalidate assumptions that
33   * this object is immutable.
34   */
35  public class ByteString {
36      /**
37       * A byte wise comparator between two {@link ByteString} objects.
38       */
39      public static final Comparator<ByteString> DEFAULT_COMPARATOR = new Comparator<ByteString>() {
40          @Override
41          public int compare(ByteString o1, ByteString o2) {
42              if (o1 == o2) {
43                  return 0;
44              }
45  
46              int result;
47              int length1 = o1.length();
48              int length2 = o2.length();
49              int minLength = Math.min(length1, length2);
50              for (int i = o1.offset, j = o2.offset; i < minLength; i++, j++) {
51                  result = o1.value[i] - o2.value[j];
52                  if (result != 0) {
53                      return result;
54                  }
55              }
56  
57              return length1 - length2;
58          }
59      };

124     /**
125      * Construct a new {@link BinaryString} object from a {@code byte[]} array.
126      * @param copy {@code true} then a copy of the memory will be made. {@code false} the underlying memory
127      * will be shared.
128      */
129     public ByteString(byte[] value, int start, int length, boolean copy) {
130         if (start < 0 || start > checkNotNull(value, "value").length - length) {
131             throw new IndexOutOfBoundsException("expected: " + "0 <= start(" + start + ") <= start + length(" + length
132                             + ") <= " + "value.length(" + value.length + ')');
133         }
134 
135         if (copy) {
136             this.value = Arrays.copyOfRange(value, start, start + length);
137             offset = 0;
138             this.length = length;
139         } else {
140             this.value = value;
141             this.offset = start;
142             this.length = length;
143         }
144     }

146     /**
147      * Create a new object which is equal to {@code value}.
148      * @param value The object to replicate.
149      * @param copy {@code true} mean the underlying storage will be copied.
150      * {@code false} means the underlying storage will be shared.
151      */
152     public ByteString(ByteString value, boolean copy) {
153         checkNotNull(value, "value");
154         this.length = value.length();
155         this.hash = value.hash;
156         if (copy) {
157             this.value = new byte[length];
158             System.arraycopy(value.array(), value.arrayOffset(), this.value, 0, length);
159             this.offset = 0;
160         } else {
161             this.value = value.array();
162             this.offset = value.offset;
163         }
164     }

184     /**
185      * Initialize a {@link ByteString} based upon the underlying storage from {@link value}.
186      * There is a potential to share the underlying array storage if {@link ByteBuffer#hasArray()} is {@code true}.
187      * if {@code copy} is {@code true} a copy will be made of the memory.
188      * if {@code copy} is {@code false} the underlying storage will be shared, if possible.
189      */
190     public ByteString(ByteBuffer value, int start, int length, boolean copy) {
191         if (start < 0 || length > checkNotNull(value, "value").capacity() - start) {
192             throw new IndexOutOfBoundsException("expected: " + "0 <= start(" + start + ") <= start + length(" + length
193                             + ") <= " + "value.capacity(" + value.capacity() + ')');
194         }
195 
196         if (value.hasArray()) {
197             if (copy) {
198                 final int bufferOffset = value.arrayOffset() + start;
199                 this.value = Arrays.copyOfRange(value.array(), bufferOffset, bufferOffset + length);
200                 offset = 0;
201                 this.length = length;
202             } else {
203                 this.value = value.array();
204                 this.offset = start;
205                 this.length = length;
206             }
207         } else {
208             this.value = new byte[length];
209             int oldPos = value.position();
210             value.get(this.value, 0, length);
211             value.position(oldPos);
212             this.offset = 0;
213             this.length = length;
214         }
215     }

224     /**
225      * Create a copy of {@link value} into a {@link ByteString} using the encoding type of {@code charset}.
226      * The copy will start at index {@code start} and copy {@code length} bytes.
227      */
228     public ByteString(char[] value, Charset charset, int start, int length) {
229         if (start < 0 || length > checkNotNull(value, "value").length - start) {
230             throw new IndexOutOfBoundsException("expected: " + "0 <= start(" + start + ") <= start + length(" + length
231                     + ") <= " + "length(" + length + ')');
232         }
233 
234         CharBuffer cbuf = CharBuffer.wrap(value, start, start + length);
235         CharsetEncoder encoder = CharsetUtil.getEncoder(charset);
236         ByteBuffer nativeBuffer = ByteBuffer.allocate((int) (encoder.maxBytesPerChar() * length));
237         encoder.encode(cbuf, nativeBuffer, true);
238         final int bufferOffset = nativeBuffer.arrayOffset();
239         this.value = Arrays.copyOfRange(nativeBuffer.array(), bufferOffset, bufferOffset + nativeBuffer.position());
240         this.offset = 0;
241         this.length =  this.value.length;
242     }

281     /**
282      * Iterates over the specified area of this buffer with the specified {@code processor} in ascending order.
283      * (i.e. {@code index}, {@code (index + 1)},  .. {@code (index + length - 1)}).
284      *
285      * @return {@code -1} if the processor iterated to or beyond the end of the specified area.
286      *         The last-visited index If the {@link ByteProcessor#process(byte)} returned {@code false}.
287      */
288     public final int forEachByte(int index, int length, ByteProcessor visitor) throws Exception {
289         if (index < 0 || length > length() - index) {
290             throw new IndexOutOfBoundsException("expected: " + "0 <= index(" + index + ") <= start + length(" + length
291                     + ") <= " + "length(" + length() + ')');
292         }
293 
294         final int len = offset + length;
295         for (int i = offset + index; i < len; ++i) {
296             if (!visitor.process(value[i])) {
297                 return i - offset;
298             }
299         }
300         return -1;
301     }

305      *
306      * @return {@code -1} if the processor iterated to or beyond the beginning of the readable bytes.
307      *         The last-visited index If the {@link ByteProcessor#process(byte)} returned {@code false}.
308      */
309     public final int forEachByteDesc(ByteProcessor visitor) throws Exception {
310         return forEachByteDesc(0, length(), visitor);

313     /**
314      * Iterates over the specified area of this buffer with the specified {@code processor} in descending order.
315      * (i.e. {@code (index + length - 1)}, {@code (index + length - 2)}, ... {@code index}).
316      *
317      * @return {@code -1} if the processor iterated to or beyond the beginning of the specified area.
318      *         The last-visited index If the {@link ByteProcessor#process(byte)} returned {@code false}.
319      */
320     public final int forEachByteDesc(int index, int length, ByteProcessor visitor) throws Exception {
321         if (index < 0 || length > length() - index) {
322             throw new IndexOutOfBoundsException("expected: " + "0 <= index(" + index + ") <= start + length(" + length
323                     + ") <= " + "length(" + length() + ')');
324         }
325 
326         final int end = offset + index;
327         for (int i = offset + index + length - 1; i >= end; --i) {
328             if (!visitor.process(value[i])) {
329                 return i - offset;
330             }
331         }
332         return -1;
333     }

335     public final byte byteAt(int index) {
336         // We must do a range check here to enforce the access does not go outside our sub region of the array.
337         // We rely on the array access itself to pick up the array out of bounds conditions
338         if (index < 0 || index >= length) {
339             throw new IndexOutOfBoundsException("index: " + index + " must be in the range [0," + length + ")");
340         }
341         return value[index + offset];
342     }

420     @Override
421     public int hashCode() {
422         int h = hash;
423         if (h == 0) {
424             final int end = offset + length;
425             for (int i = offset; i < end; ++i) {
426                 h = h * HASH_CODE_PRIME ^ value[i] & HASH_CODE_PRIME;
427             }
428 
429             hash = h;
430         }
431         return hash;
432     }

469      * Either copy or share a subset of underlying sub-sequence of bytes.
470      * @param start the offset of the first character (inclusive).
471      * @param end The index to stop at (exclusive).
472      * @param copy If {@code true} then a copy of the underlying storage will be made.
473      * If {@code false} then the underlying storage will be shared.
474      * @param factory The factory used to generate a new {@link ByteString} object.
475      * @return a new string containing the characters from start to the end of the string.
476      * @throws IndexOutOfBoundsException if {@code start < 0} or {@code start > length()}.

478     protected ByteString subSequence(int start, int end, boolean copy, ByteStringFactory factory) {
479         if (start < 0 || start > end || end > length()) {
480             throw new IndexOutOfBoundsException("expected: 0 <= start(" + start + ") <= end (" + end + ") <= length("
481                             + length() + ')');
482         }
483 
484         if (start == 0 && end == length()) {
485             return this;
486         }
487 
488         if (end == start) {
489             return EMPTY_STRING;
490         }
491 
492         return factory.newInstance(value, start + offset, end - start, copy);
493     }

507     public final int parseAsciiInt(int start, int end, int radix) {
508         if (radix < Character.MIN_RADIX || radix > Character.MAX_RADIX) {
509             throw new NumberFormatException();
510         }
511 
512         if (start == end) {
513             throw new NumberFormatException();
514         }
515 
516         int i = start;
517         boolean negative = byteAt(i) == '-';
518         if (negative && ++i == end) {
519             throw new NumberFormatException(subSequence(start, end, false).toString());
520         }
521 
522         return parseAsciiInt(i, end, radix, negative);
523     }

525     private int parseAsciiInt(int start, int end, int radix, boolean negative) {
526         int max = Integer.MIN_VALUE / radix;
527         int result = 0;
528         int currOffset = start;
529         while (currOffset < end) {
530             int digit = Character.digit((char) (value[currOffset++ + offset] & 0xFF), radix);
531             if (digit == -1) {
532                 throw new NumberFormatException(subSequence(start, end, false).toString());
533             }
534             if (max > result) {
535                 throw new NumberFormatException(subSequence(start, end, false).toString());
536             }
537             int next = result * radix - digit;
538             if (next > result) {
539                 throw new NumberFormatException(subSequence(start, end, false).toString());
540             }
541             result = next;
542         }
543         if (!negative) {
544             result = -result;
545             if (result < 0) {
546                 throw new NumberFormatException(subSequence(start, end, false).toString());
547             }
548         }
549         return result;
550     }

613     public char parseChar(int start) {
614         if (start + 1 >= length()) {
615             throw new IndexOutOfBoundsException("2 bytes required to convert to character. index " +
616                     start + " would go out of bounds.");
617         }
618         final int startWithOffset = start + offset;
619         return (char) (((value[startWithOffset] & 0xFF) << 8) | (value[startWithOffset + 1] & 0xFF));
620     }

659     @Override
660     public boolean equals(Object obj) {
661         if (!(obj instanceof ByteString)) {
662             return false;
663         }
664         if (this == obj) {
665             return true;
666         }
667
