23  /**
24   * A hash map implementation of {@link IntObjectMap} that uses open addressing for keys. To minimize
25   * the memory footprint, this class uses open addressing rather than chaining. Collisions are
26   * resolved using double hashing.
27   *
28   * @param <V> The value type stored in the map.
29   */
30  public class IntObjectHashMap<V> implements IntObjectMap<V>, Iterable<IntObjectMap.Entry<V>> {
31  
32      /** State indicating that a slot is available.*/
33      private static final byte AVAILABLE = 0;
34  
35      /** State indicating that a slot is occupied. */
36      private static final byte OCCUPIED = 1;
37  
38      /** State indicating that a slot was removed. */
39      private static final byte REMOVED = 2;
40  
41      /** Default initial capacity. Used if not specified in the constructor */
42      private static final int DEFAULT_CAPACITY = 11;
43  
44      /** Default load factor. Used if not specified in the constructor */
45      private static final float DEFAULT_LOAD_FACTOR = 0.5f;
46  
47      /** The maximum number of elements allowed without allocating more space. */
48      private int maxSize;
49  
50      /** The load factor for the map. Used to calculate {@link #maxSize}. */
51      private final float loadFactor;

53      private byte[] states;
54      private int[] keys;
55      private V[] values;
56      private int size;
57      private int available;

59      public IntObjectHashMap() {
60          this(DEFAULT_CAPACITY, DEFAULT_LOAD_FACTOR);
61      }

63      public IntObjectHashMap(int initialCapacity) {
64          this(initialCapacity, DEFAULT_LOAD_FACTOR);

67      public IntObjectHashMap(int initialCapacity, float loadFactor) {
68          if (initialCapacity < 1) {
69              throw new IllegalArgumentException("initialCapacity must be >= 1");
70          }
71          if (loadFactor <= 0.0f) {
72              throw new IllegalArgumentException("loadFactor must be > 0");
73          }
74  
75          this.loadFactor = loadFactor;
76  
77          // Adjust the initial capacity if necessary.
78          initialCapacity = adjustCapacity(initialCapacity);
79  
80          // Allocate the arrays.
81          states = new byte[initialCapacity];
82          keys = new int[initialCapacity];
83          @SuppressWarnings({ "unchecked", "SuspiciousArrayCast" })
84          V[] temp = (V[]) new Object[initialCapacity];
85          values = temp;
86  
87          // Initialize the maximum size value.
88          maxSize = calcMaxSize(initialCapacity);
89  
90          // Initialize the available element count
91          available = initialCapacity - size;
92      }

94      @Override
95      public V get(int key) {
96          int index = indexOf(key);
97          return index < 0 ? null : values[index];
98      }

100     @Override
101     public V put(int key, V value) {
102         int hash = hash(key);
103         int capacity = capacity();
104         int index = hash % capacity;
105         int increment = 1 + hash % (capacity - 2);
106         final int startIndex = index;
107         int firstRemovedIndex = -1;
108         do {
109             switch (states[index]) {
110                 case AVAILABLE:
111                     // We only stop probing at a AVAILABLE node, since the value may still exist
112                     // beyond
113                     // a REMOVED node.
114                     if (firstRemovedIndex != -1) {
115                         // We encountered a REMOVED node prior. Store the entry there so that
116                         // retrieval
117                         // will be faster.
118                         insertAt(firstRemovedIndex, key, value);
119                         return null;
120                     }
121 
122                     // No REMOVED node, just store the entry here.
123                     insertAt(index, key, value);
124                     return null;
125                 case OCCUPIED:
126                     if (keys[index] == key) {
127                         V previousValue = values[index];
128                         insertAt(index, key, value);
129                         return previousValue;
130                     }
131                     break;
132                 case REMOVED:
133                     // Check for first removed index.
134                     if (firstRemovedIndex == -1) {
135                         firstRemovedIndex = index;
136                     }
137                     break;
138                 default:
139                     throw new AssertionError("Invalid state: " + states[index]);
140             }
141 
142             // REMOVED or OCCUPIED but wrong key, keep probing ...
143             index += increment;
144             if (index >= capacity) {
145                 // Handle wrap-around by decrement rather than mod.
146                 index -= capacity;
147             }
148         } while (index != startIndex);
149 
150         if (firstRemovedIndex == -1) {
151             // Should never happen.
152             throw new AssertionError("Unable to insert");
153         }
154 
155         // Never found a AVAILABLE slot, just use the first REMOVED.
156         insertAt(firstRemovedIndex, key, value);
157         return null;
158     }

160     @Override
161     public void putAll(IntObjectMap<V> sourceMap) {
162         if (sourceMap instanceof IntObjectHashMap) {
163             // Optimization - iterate through the arrays.
164             IntObjectHashMap<V> source = (IntObjectHashMap<V>) sourceMap;
165             int i = -1;
166             while ((i = source.nextEntryIndex(i + 1)) >= 0) {
167                 put(source.keys[i], source.values[i]);
168             }
169             return;
170         }
171 
172         // Otherwise, just add each entry.
173         for (Entry<V> entry : sourceMap.entries()) {
174             put(entry.key(), entry.value());
175         }
176     }

213     @Override
214     public boolean containsValue(V value) {
215         int i = -1;
216         while ((i = nextEntryIndex(i + 1)) >= 0) {
217             V next = values[i];
218             if (value == next || value != null && value.equals(next)) {
219                 return true;
220             }
221         }
222         return false;

250     /**
251      * Copies the occupied entries from the source to the target array.
252      */
253     private void copyEntries(Object sourceArray, Object targetArray) {
254         int sourceIx = -1;
255         int targetIx = 0;
256         while ((sourceIx = nextEntryIndex(sourceIx + 1)) >= 0) {
257             Object obj = Array.get(sourceArray, sourceIx);
258             Array.set(targetArray, targetIx++, obj);
259         }
260     }

262     /**
263      * Locates the index for the given key. This method probes using double hashing.
264      *
265      * @param key the key for an entry in the map.
266      * @return the index where the key was found, or {@code -1} if no entry is found for that key.
267      */
268     private int indexOf(int key) {
269         int hash = hash(key);
270         int capacity = capacity();
271         int increment = 1 + hash % (capacity - 2);
272         int index = hash % capacity;
273         int startIndex = index;
274         do {
275             switch(states[index]) {
276                 case AVAILABLE:
277                     // It's available, so no chance that this value exists anywhere in the map.
278                     return -1;
279                 case OCCUPIED:
280                     if (key == keys[index]) {
281                         // Found it!
282                         return index;
283                     }
284                     break;
285                 default:
286                     break;
287             }
288 
289             // REMOVED or OCCUPIED but wrong key, keep probing ...
290             index += increment;
291             if (index >= capacity) {
292                 // Handle wrap-around by decrement rather than mod.
293                 index -= capacity;
294             }
295         } while (index != startIndex);
296 
297         // Got back to the beginning. Not found.
298         return -1;
299     }

308     /**
309      * Creates a hash value for the given key.
310      */
311     private static int hash(int key) {
312         // Just make sure the integer is positive.
313         return key & Integer.MAX_VALUE;
314     }

316     /**
317      * Performs an insert of the key/value at the given index position. If necessary, performs a
318      * rehash of the map.
319      *
320      * @param index the index at which to insert the key/value
321      * @param key the entry key
322      * @param value the entry value
323      */
324     private void insertAt(int index, int key, V value) {
325         byte state = states[index];
326         if (state != OCCUPIED) {
327             // Added a new mapping, increment the size.
328             size++;
329 
330             if (state == AVAILABLE) {
331                 // Consumed a OCCUPIED slot, decrement the number of available slots.
332                 available--;
333             }
334         }
335 
336         keys[index] = key;
337         values[index] = value;
338         states[index] = OCCUPIED;
339 
340         if (size > maxSize) {
341             // Need to grow the arrays.
342             rehash(adjustCapacity(capacity() * 2));
343         } else if (available == 0) {
344             // Open addressing requires that we have at least 1 slot available. Need to refresh
345             // the arrays to clear any removed elements.
346             rehash(capacity());
347         }
348     }

350     /**
351      * Adjusts the given capacity value to ensure that it's odd. Even capacities can break probing.
352      * TODO: would be better to ensure it's prime as well.
353      */
354     private static int adjustCapacity(int capacity) {
355         return capacity | 1;
356     }
357 
358     /**
359      * Marks the entry at the given index position as {@link #REMOVED} and sets the value to
360      * {@code null}.
361      * <p>
362      * TODO: consider performing re-compaction.
363      *
364      * @param index the index position of the element to remove.
365      */
366     private void removeAt(int index) {
367         if (states[index] == OCCUPIED) {
368             size--;
369         }
370         states[index] = REMOVED;
371         values[index] = null;
372     }

374     /**
375      * Calculates the maximum size allowed before rehashing.
376      */
377     private int calcMaxSize(int capacity) {
378         // Clip the upper bound so that there will always be at least one
379         // available slot.
380         int upperBound = capacity - 1;
381         return Math.min(upperBound, (int) (capacity * loadFactor));
382     }

384     /**
385      * Rehashes the map for the given capacity.
386      *
387      * @param newCapacity the new capacity for the map.
388      */
389     private void rehash(int newCapacity) {
390         int oldCapacity = capacity();
391         int[] oldKeys = keys;
392         V[] oldVals = values;
393         byte[] oldStates = states;
394 
395         // New states array is automatically initialized to AVAILABLE (i.e. 0 == AVAILABLE).
396         states = new byte[newCapacity];
397         keys = new int[newCapacity];
398         @SuppressWarnings({ "unchecked", "SuspiciousArrayCast" })
399         V[] temp = (V[]) new Object[newCapacity];
400         values = temp;
401 
402         size = 0;
403         available = newCapacity;
404         maxSize = calcMaxSize(newCapacity);
405 
406         // Insert the new states.
407         for (int i = 0; i < oldCapacity; ++i) {
408             if (oldStates[i] == OCCUPIED) {
409                 put(oldKeys[i], oldVals[i]);
410             }
411         }
412     }

414     /**
415      * Returns the next index of the next entry in the map.
416      *
417      * @param index the index at which to begin the search.
418      * @return the index of the next entry, or {@code -1} if not found.
419      */
420     private int nextEntryIndex(int index) {
421         int capacity = capacity();
422         for (; index < capacity; ++index) {
423             if (states[index] == OCCUPIED) {
424                 return index;
425             }
426         }
427         return -1;
428     }

430     /**
431      * Iterator for traversing the entries in this map.
432      */
433     private final class IteratorImpl implements Iterator<Entry<V>> {
434         int prevIndex;
435         int nextIndex;
436 
437         IteratorImpl() {
438             prevIndex = -1;
439             nextIndex = nextEntryIndex(0);
440         }
441 
442         @Override
443         public boolean hasNext() {
444             return nextIndex >= 0;
445         }
446 
447         @Override
448         public Entry<V> next() {
449             if (!hasNext()) {
450                 throw new NoSuchElementException();
451             }
452 
453             prevIndex = nextIndex;
454             nextIndex = nextEntryIndex(nextIndex + 1);
455             return new EntryImpl(prevIndex);
456         }
457 
458         @Override
459         public void remove() {
460             if (prevIndex < 0) {
461                 throw new IllegalStateException("Next must be called before removing.");
462             }
463             removeAt(prevIndex);
464             prevIndex = -1;
465         }
466     }
