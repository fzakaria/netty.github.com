128     void addMessage(Object msg, ChannelPromise promise) {
129         int size = channel.estimatorHandle().size(msg);
130         if (size < 0) {
131             size = 0;
132         }
133 
134         Entry e = buffer[tail++];
135         e.msg = msg;
136         e.pendingSize = size;
137         e.promise = promise;
138         e.total = total(msg);
139 
140         tail &= buffer.length - 1;
141 
142         if (tail == flushed) {
143             addCapacity();
144         }
145 
146         // increment pending bytes after adding message to the unflushed arrays.
147         // See https://github.com/netty/netty/issues/1619
148         incrementPendingOutboundBytes(size);
149     }

151     private void addCapacity() {
152         int p = flushed;
153         int n = buffer.length;
154         int r = n - p; // number of elements to the right of p
155         int s = size();
156 
157         int newCapacity = n << 1;
158         if (newCapacity < 0) {
159             throw new IllegalStateException();
160         }
161 
162         Entry[] e = new Entry[newCapacity];
163         System.arraycopy(buffer, p, e, 0, r);
164         System.arraycopy(buffer, 0, e, r, p);
165         for (int i = n; i < e.length; i++) {
166             e[i] = new Entry();
167         }
168 
169         buffer = e;
170         flushed = 0;
171         unflushed = s;
172         tail = n;
173     }

175     void addFlush() {
176         unflushed = tail;
177 
178         final int mask = buffer.length - 1;
179         int i = flushed;
180         while (i != unflushed && buffer[i].msg != null) {
181             Entry entry = buffer[i];
182             if (!entry.promise.setUncancellable()) {
183                 // Was cancelled so make sure we free up memory and notify about the freed bytes
184                 int pending = entry.cancel();
185                 decrementPendingOutboundBytes(pending);
186             }
187             i = i + 1 & mask;
188         }
189     }

191     /**
192      * Increment the pending bytes which will be written at some point.
193      * This method is thread-safe!
194      */
195     void incrementPendingOutboundBytes(int size) {
196         // Cache the channel and check for null to make sure we not produce a NPE in case of the Channel gets
197         // recycled while process this method.
198         Channel channel = this.channel;
199         if (size == 0 || channel == null) {
200             return;
201         }
202 
203         long oldValue = totalPendingSize;
204         long newWriteBufferSize = oldValue + size;
205         while (!TOTAL_PENDING_SIZE_UPDATER.compareAndSet(this, oldValue, newWriteBufferSize)) {
206             oldValue = totalPendingSize;
207             newWriteBufferSize = oldValue + size;
208         }
209 
210         int highWaterMark = channel.config().getWriteBufferHighWaterMark();
211 
212         if (newWriteBufferSize > highWaterMark) {
213             if (WRITABLE_UPDATER.compareAndSet(this, 1, 0)) {
214                 channel.pipeline().fireChannelWritabilityChanged();
215             }
216         }
217     }
218 
219     /**
220      * Decrement the pending bytes which will be written at some point.
221      * This method is thread-safe!
222      */
223     void decrementPendingOutboundBytes(int size) {
224         // Cache the channel and check for null to make sure we not produce a NPE in case of the Channel gets
225         // recycled while process this method.
226         Channel channel = this.channel;
227         if (size == 0 || channel == null) {
228             return;
229         }
230 
231         long oldValue = totalPendingSize;
232         long newWriteBufferSize = oldValue - size;
233         while (!TOTAL_PENDING_SIZE_UPDATER.compareAndSet(this, oldValue, newWriteBufferSize)) {
234             oldValue = totalPendingSize;
235             newWriteBufferSize = oldValue - size;
236         }
237 
238         int lowWaterMark = channel.config().getWriteBufferLowWaterMark();
239 
240         if (newWriteBufferSize == 0 || newWriteBufferSize < lowWaterMark) {
241             if (WRITABLE_UPDATER.compareAndSet(this, 0, 1)) {
242                 channel.pipeline().fireChannelWritabilityChanged();
243             }
244         }
245     }

247     private static long total(Object msg) {
248         if (msg instanceof ByteBuf) {
249             return ((ByteBuf) msg).readableBytes();
250         }
251         if (msg instanceof FileRegion) {
252             return ((FileRegion) msg).count();
253         }
254         if (msg instanceof ByteBufHolder) {
255             return ((ByteBufHolder) msg).content().readableBytes();
256         }
257         return -1;
258     }

264     public Object current(boolean preferDirect) {
265         if (isEmpty()) {
266             return null;
267         } else {
268             // TODO: Think of a smart way to handle ByteBufHolder messages
269             Entry entry = buffer[flushed];
270             Object msg = entry.msg;
271             if (threadLocalDirectBufferSize <= 0 || !preferDirect) {
272                 return msg;
273             }
274             if (msg instanceof ByteBuf) {
275                 ByteBuf buf = (ByteBuf) msg;
276                 if (buf.isDirect()) {
277                     return buf;
278                 } else {
279                     int readableBytes = buf.readableBytes();
280                     if (readableBytes == 0) {
281                         return buf;
282                     }
283 
284                     // Non-direct buffers are copied into JDK's own internal direct buffer on every I/O.
285                     // We can do a better job by using our pooled allocator. If the current allocator does not
286                     // pool a direct buffer, we use a ThreadLocal based pool.
287                     ByteBufAllocator alloc = channel.alloc();
288                     ByteBuf directBuf;
289                     if (alloc.isDirectBufferPooled()) {
290                         directBuf = alloc.directBuffer(readableBytes);
291                     } else {
292                         directBuf = ThreadLocalPooledByteBuf.newInstance();
293                     }
294                     directBuf.writeBytes(buf, buf.readerIndex(), readableBytes);
295                     current(directBuf);
296                     return directBuf;
297                 }
298             }
299             return msg;
300         }
301     }

323     public boolean remove() {
324         if (isEmpty()) {
325             return false;
326         }
327 
328         Entry e = buffer[flushed];
329         Object msg = e.msg;
330         if (msg == null) {
331             return false;
332         }
333 
334         ChannelPromise promise = e.promise;
335         int size = e.pendingSize;
336 
337         e.clear();
338 
339         flushed = flushed + 1 & buffer.length - 1;
340 
341         if (!e.cancelled) {
342             // only release message, notify and decrement if it was not canceled before.
343             safeRelease(msg);
344             safeSuccess(promise);
345             decrementPendingOutboundBytes(size);
346         }
347 
348         return true;
349     }

351     public boolean remove(Throwable cause) {
352         if (isEmpty()) {
353             return false;
354         }
355 
356         Entry e = buffer[flushed];
357         Object msg = e.msg;
358         if (msg == null) {
359             return false;
360         }
361 
362         ChannelPromise promise = e.promise;
363         int size = e.pendingSize;
364 
365         e.clear();
366 
367         flushed = flushed + 1 & buffer.length - 1;
368 
369         if (!e.cancelled) {
370             // only release message, fail and decrement if it was not canceled before.
371             safeRelease(msg);
372 
373             safeFail(promise, cause);
374             decrementPendingOutboundBytes(size);
375         }
376 
377         return true;
378     }

380     /**
381      * Returns an array of direct NIO buffers if the currently pending messages are made of {@link ByteBuf} only.
382      * {@code null} is returned otherwise.  If this method returns a non-null array, {@link #nioBufferCount()} and
383      * {@link #nioBufferSize()} will return the number of NIO buffers in the returned array and the total number
384      * of readable bytes of the NIO buffers respectively.
385      * <p>
386      * Note that the returned array is reused and thus should not escape
387      * {@link AbstractChannel#doWrite(ChannelOutboundBuffer)}.
388      * Refer to {@link NioSocketChannel#doWrite(ChannelOutboundBuffer)} for an example.
389      * </p>
390      */
391     public ByteBuffer[] nioBuffers() {
392         long nioBufferSize = 0;
393         int nioBufferCount = 0;
394         final int mask = buffer.length - 1;
395         final ByteBufAllocator alloc = channel.alloc();
396         ByteBuffer[] nioBuffers = this.nioBuffers;
397         Object m;
398         int i = flushed;
399         while (i != unflushed && (m = buffer[i].msg) != null) {
400             if (!(m instanceof ByteBuf)) {
401                 this.nioBufferCount = 0;
402                 this.nioBufferSize = 0;
403                 return null;
404             }
405 
406             Entry entry = buffer[i];
407 
408             if (!entry.cancelled) {
409                 ByteBuf buf = (ByteBuf) m;
410                 final int readerIndex = buf.readerIndex();
411                 final int readableBytes = buf.writerIndex() - readerIndex;
412 
413                 if (readableBytes > 0) {
414                     nioBufferSize += readableBytes;
415                     int count = entry.count;
416                     if (count == -1) {
417                         //noinspection ConstantValueVariableUse
418                         entry.count = count =  buf.nioBufferCount();
419                     }
420                     int neededSpace = nioBufferCount + count;
421                     if (neededSpace > nioBuffers.length) {
422                         this.nioBuffers = nioBuffers =
423                                 expandNioBufferArray(nioBuffers, neededSpace, nioBufferCount);
424                     }
425                     if (buf.isDirect() || threadLocalDirectBufferSize <= 0) {
426                         if (count == 1) {
427                             ByteBuffer nioBuf = entry.buf;
428                             if (nioBuf == null) {
429                                 // cache ByteBuffer as it may need to create a new ByteBuffer instance if its a
430                                 // derived buffer
431                                 entry.buf = nioBuf = buf.internalNioBuffer(readerIndex, readableBytes);
432                             }
433                             nioBuffers[nioBufferCount ++] = nioBuf;
434                         } else {
435                             ByteBuffer[] nioBufs = entry.buffers;
436                             if (nioBufs == null) {
437                                 // cached ByteBuffers as they may be expensive to create in terms
438                                 // of Object allocation
439                                 entry.buffers = nioBufs = buf.nioBuffers();
440                             }
441                             nioBufferCount = fillBufferArray(nioBufs, nioBuffers, nioBufferCount);
442                         }
443                     } else {
444                         nioBufferCount = fillBufferArrayNonDirect(entry, buf, readerIndex,
445                                 readableBytes, alloc, nioBuffers, nioBufferCount);
446                     }
447                 }
448             }
449 
450             i = i + 1 & mask;
451         }
452         this.nioBufferCount = nioBufferCount;
453         this.nioBufferSize = nioBufferSize;
454 
455         return nioBuffers;
456     }

458     private static int fillBufferArray(ByteBuffer[] nioBufs, ByteBuffer[] nioBuffers, int nioBufferCount) {
459         for (ByteBuffer nioBuf: nioBufs) {
460             if (nioBuf == null) {
461                 break;
462             }
463             nioBuffers[nioBufferCount ++] = nioBuf;
464         }
465         return nioBufferCount;
466     }

468     private static int fillBufferArrayNonDirect(Entry entry, ByteBuf buf, int readerIndex, int readableBytes,
469                                       ByteBufAllocator alloc, ByteBuffer[] nioBuffers, int nioBufferCount) {
470         ByteBuf directBuf;
471         if (alloc.isDirectBufferPooled()) {
472             directBuf = alloc.directBuffer(readableBytes);
473         } else {
474             directBuf = ThreadLocalPooledByteBuf.newInstance();
475         }
476         directBuf.writeBytes(buf, readerIndex, readableBytes);
477         buf.release();
478         entry.msg = directBuf;
479         // cache ByteBuffer
480         ByteBuffer nioBuf = entry.buf = directBuf.internalNioBuffer(0, readableBytes);
481         entry.count = 1;
482         nioBuffers[nioBufferCount ++] = nioBuf;
483         return nioBufferCount;
484     }

486     private static ByteBuffer[] expandNioBufferArray(ByteBuffer[] array, int neededSpace, int size) {
487         int newCapacity = array.length;
488         do {
489             // double capacity until it is big enough
490             // See https://github.com/netty/netty/issues/1890
491             newCapacity <<= 1;
492 
493             if (newCapacity < 0) {
494                 throw new IllegalStateException();
495             }
496 
497         } while (neededSpace > newCapacity);
498 
499         ByteBuffer[] newArray = new ByteBuffer[newCapacity];
500         System.arraycopy(array, 0, newArray, 0, size);
501 
502         return newArray;
503     }

525     void failFlushed(Throwable cause) {
526         // Make sure that this method does not reenter.  A listener added to the current promise can be notified by the
527         // current thread in the tryFailure() call of the loop below, and the listener can trigger another fail() call
528         // indirectly (usually by closing the channel.)
529         //
530         // See https://github.com/netty/netty/issues/1501
531         if (inFail) {
532             return;
533         }
534 
535         try {
536             inFail = true;
537             for (;;) {
538                 if (!remove(cause)) {
539                     break;
540                 }
541             }
542         } finally {
543             inFail = false;
544         }
545     }

547     void close(final ClosedChannelException cause) {
548         if (inFail) {
549             channel.eventLoop().execute(new Runnable() {
550                 @Override
551                 public void run() {
552                     close(cause);
553                 }
554             });
555             return;
556         }
557 
558         inFail = true;
559 
560         if (channel.isOpen()) {
561             throw new IllegalStateException("close() must be invoked after the channel is closed.");
562         }
563 
564         if (!isEmpty()) {
565             throw new IllegalStateException("close() must be invoked after all flushed writes are handled.");
566         }
567 
568         // Release all unflushed messages.
569         final int unflushedCount = tail - unflushed & buffer.length - 1;
570         try {
571             for (int i = 0; i < unflushedCount; i++) {
572                 Entry e = buffer[unflushed + i & buffer.length - 1];
573 
574                 // Just decrease; do not trigger any events via decrementPendingOutboundBytes()
575                 int size = e.pendingSize;
576                 long oldValue = totalPendingSize;
577                 long newWriteBufferSize = oldValue - size;
578                 while (!TOTAL_PENDING_SIZE_UPDATER.compareAndSet(this, oldValue, newWriteBufferSize)) {
579                     oldValue = totalPendingSize;
580                     newWriteBufferSize = oldValue - size;
581                 }
582 
583                 e.pendingSize = 0;
584                 if (!e.cancelled) {
585                     safeRelease(e.msg);
586                     safeFail(e.promise, cause);
587                 }
588                 e.msg = null;
589                 e.promise = null;
590             }
591         } finally {
592             tail = unflushed;
593             inFail = false;
594         }
595 
596         recycle();
597     }

619     public void recycle() {
620         if (buffer.length > INITIAL_CAPACITY) {
621             Entry[] e = new Entry[INITIAL_CAPACITY];
622             System.arraycopy(buffer, 0, e, 0, INITIAL_CAPACITY);
623             buffer = e;
624         }
625 
626         if (nioBuffers.length > INITIAL_CAPACITY) {
627             nioBuffers = new ByteBuffer[INITIAL_CAPACITY];
628         } else {
629             // null out the nio buffers array so the can be GC'ed
630             // https://github.com/netty/netty/issues/1763
631             Arrays.fill(nioBuffers, null);
632         }
633 
634         // reset flushed, unflushed and tail
635         // See https://github.com/netty/netty/issues/1772
636         flushed = 0;
637         unflushed = 0;
638         tail = 0;
639 
640         // Set the channel to null so it can be GC'ed ASAP
641         channel = null;
642 
643         RECYCLER.recycle(this, handle);
644     }

650     private static final class Entry {
651         Object msg;
652         ByteBuffer[] buffers;
653         ByteBuffer buf;
654         ChannelPromise promise;
655         long progress;
656         long total;
657         int pendingSize;
658         int count = -1;
659         boolean cancelled;
660 
661         public int cancel() {
662             if (!cancelled) {
663                 cancelled = true;
664                 int pSize = pendingSize;
665 
666                 // release message and replace with an empty buffer
667                 safeRelease(msg);
668                 msg = Unpooled.EMPTY_BUFFER;
669 
670                 pendingSize = 0;
671                 total = 0;
672                 progress = 0;
673                 buffers = null;
674                 buf = null;
675                 return pSize;
676             }
677             return 0;
678         }
679 
680         public void clear() {
681             buffers = null;
682             buf = null;
683             msg = null;
684             promise = null;
685             progress = 0;
686             total = 0;
687             pendingSize = 0;
688             count = -1;
689             cancelled = false;
690         }
